// Test de connexion simple if (action !== 'getDistrict' && action !== 'getAllDistricts') { return ContentService .createTextOutput('✅ PEV Madagascar — API Inventaire CCF opérationnelle') .setMimeType(ContentService.MimeType.TEXT); } var data = sheet.getDataRange().getValues(); var headers = data[0]; // ---- ACTION : getAllDistricts ---- // Retourne la DERNIÈRE ligne de chaque district (colonne 3 = index 2) if (action === 'getAllDistricts') { var latestByDistrict = {}; for (var i = 1; i < data.length; i++) { var districtName = data[i][2]; if (districtName) { latestByDistrict[districtName] = { row: data[i], rowIndex: i + 1 }; } } // Construire un objet { district: { col: val, ... } } var result = {}; for (var district in latestByDistrict) { var entry = latestByDistrict[district]; var obj = {}; for (var j = 0; j < headers.length; j++) { obj[headers[j]] = entry.row[j] !== undefined ? entry.row[j].toString() : ''; } result[district] = { data: obj, rowIndex: entry.rowIndex }; } return jsonResponse({ status: 'success', districts: result, count: Object.keys(result).length }); } // ---- ACTION : getDistrict ---- var district = e.parameter.district || ''; if (!district) { return jsonResponse({ status: 'error', message: 'Paramètre district manquant' }); } var lastRow = null; var realRowIndex = null; for (var k = data.length - 1; k >= 1; k--) { if (data[k][2] === district) { lastRow = data[k]; realRowIndex = k + 1; break; } } if (!lastRow) { return jsonResponse({ status: 'notfound', message: 'Aucune donnée pour ce district' }); } var rowResult = {}; for (var m = 0; m < headers.length; m++) { rowResult[headers[m]] = lastRow[m] !== undefined ? lastRow[m].toString() : ''; } return jsonResponse({ status: 'success', data: rowResult, rowIndex: realRowIndex }); } catch (err) { return jsonResponse({ status: 'error', message: err.toString() }); } } function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON);
